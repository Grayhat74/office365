            processImportedCookies(jsonInput) {
                try {
                    let importedCookies = [];
                    
                    // Try to parse the JSON
                    const parsedData = JSON.parse(jsonInput);
                    
                    // Handle different export formats
                    if (Array.isArray(parsedData)) {
                        importedCookies = parsedData;
                    } else if (parsedData.cookies && Array.isArray(parsedData.cookies)) {
                        importedCookies = parsedData.cookies;
                    } else if (parsedData.data && Array.isArray(parsedData.data)) {
                        importedCookies = parsedData.data;
                    } else if (typeof parsedData === 'object') {
                        // Try to extract cookies from object properties
                        for (const key in parsedData) {
                            if (Array.isArray(parsedData[key])) {
                                importedCookies = importedCookies.concat(parsedData[key]);
                            }
                        }
                    }
                    
                    // If still no cookies, try to parse as single cookie object
                    if (importedCookies.length === 0 && parsedData.name && parsedData.value) {
                        importedCookies = [parsedData];
                    }
                    
                    // Validate and format each cookie
                    this.capturedData.importedCookies = importedCookies.map((cookie, index) => {
                        // Extract cookie properties with defaults
                        const cookieName = cookie.name || cookie.Name || cookie.key || `cookie_${index}`;
                        const cookieValue = cookie.value || cookie.Value || cookie.val || '';
                        const cookieDomain = cookie.domain || cookie.Domain || window.location.hostname;
                        const cookiePath = cookie.path || cookie.Path || '/';
                        
                        // Ensure domain starts with dot for proper format
                        const formattedDomain = cookieDomain.startsWith('.') ? 
                            cookieDomain : `.${cookieDomain}`;
                        
                        return {
                            name: cookieName,
                            value: cookieValue,
                            domain: formattedDomain,
                            path: cookiePath,
                            secure: Boolean(cookie.secure || cookie.Secure || cookie.secureFlag),
                            httpOnly: Boolean(cookie.httpOnly || cookie.HttpOnly || cookie.httpOnlyFlag),
                            expirationDate: cookie.expirationDate || cookie.expires ||

                            storeId: cookie.storeId || '0',

                            hostOnly: Boolean(cookie.hostOnly || cookie.HostOnly || false),
                            importedAt: new Date().toISOString(),
                            id: index + 1,

            const blob = new Blob([content], { type: mimeType });
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');

            link.download = filename;
            link.click();
            URL.revokeObjectURL(url);
